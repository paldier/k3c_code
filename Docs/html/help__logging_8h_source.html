80 static void LOGF_LOG_PRINT(const char *color_code, const char *logtype, const char *fmt, ...) {
81  char cAppendStr[256]={0};
82  va_list args;
83 #ifdef DBG_TIMESTAMP
84  struct timeval tv;
85  struct tm * timeinfo;

88  gettimeofday(&tv, NULL);
89  nowtime = tv.tv_sec;
90  timeinfo = localtime(&nowtime);
91  if(timeinfo != NULL)
92  strftime(tmbuf,50,"%d-%m-%Y %T", timeinfo);
93 #endif
94 #ifdef DBG_TIMESTAMP
95  snprintf(cAppendStr, sizeof(cAppendStr), "%s<%s> [%s] ",color_code, tmbuf, logtype);
96 #else
97  snprintf(cAppendStr, sizeof(cAppendStr), "%s[%s] ",color_code, logtype);
98 #endif
99  do{
100  printf("%s", cAppendStr);
101  va_start(args, fmt);
102  vprintf(fmt, args);
103  va_end(args);
104  printf("%s",COLOR_NRM);
105  fflush(stdout);
106  }while(0);
107 }
