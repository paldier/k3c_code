94 static inline void * HELP_CREATE_OBJ(IN uint32_t unSubOper)
95 {
96  if (OBJLIST(unSubOper))
97  {
98  ObjList *__pxTmpObj;
99  __pxTmpObj = HELP_MALLOC(sizeof(ObjList));
100  if(!__pxTmpObj)
101  {
102  LOGF_LOG_CRITICAL(" malloc failed\n");
103  return NULL;
104  }
105  INIT_LIST_HEAD(&(__pxTmpObj->xOlist));
106  return (void*)__pxTmpObj;
107  }
108  else if (OBJATTRLIST(unSubOper))
109  {
110  ObjAttrList *__pxTmpAttrObj;
111  __pxTmpAttrObj = HELP_MALLOC(sizeof(ObjAttrList));
112  if(!__pxTmpAttrObj)
113  {
114  LOGF_LOG_CRITICAL(" malloc failed\n");
115  return NULL;
116  }
117  INIT_LIST_HEAD(&(__pxTmpAttrObj->xOlist));
118  return (void*)__pxTmpAttrObj;
119 
120  }
121  else if (IS_SOPT_OBJACSATTR(unSubOper))
122  {
123  ObjACSList *__pxTmpAcsObj;
124  __pxTmpAcsObj = HELP_MALLOC(sizeof(ObjACSList));
125  if(!__pxTmpAcsObj)
126  {
127  LOGF_LOG_CRITICAL(" malloc failed\n");
128  return NULL;
129  }
130  INIT_LIST_HEAD(&(__pxTmpAcsObj->xOlist));
131  return (void*)__pxTmpAcsObj;
132  }
133  else
134  {
135  LOGF_LOG_ERROR(" Pass Proper Objlist Flag (i.e SOPT_OBJVALUE, SOPT_OBJATTR, SOPT_OBJACSATTR\n");
136  return NULL;
137  }
